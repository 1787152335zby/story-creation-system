━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
第1步：确认新设备有 Python 3.10+
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

打开终端（cmd/PowerShell），输入：

    python --version

如果输出版本号（如 Python 3.12.x），跳到第3步。
如果提示"未找到命令"，继续第2步。


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
第2步：安装 Python
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

去 https://www.python.org/downloads/ 下载 Python 3.12
安装时勾选"Add Python to PATH"

安装完成后关掉终端重新打开，输入 python --version 确认。


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
第3步：安装依赖
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

在终端里进入"创作系统"文件夹（你的U盘路径可能不同）：

    cd D:\创作系统        ← 换成你U盘的实际盘符

然后运行：

    pip install -r requirements.txt

等待安装完成，总共就5个包，几分钟就装完。


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
第4步：配置 API 密钥
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

在"创作系统"文件夹里找到 .env.example 文件，
复制一份改名为 .env（去掉 .example）。

用记事本打开 .env，填入你的 API Key（至少填一个就行）：

    如果你用 DeepSeek（推荐）：
        LLM_BACKEND=deepseek
        DEEPSEEK_API_KEY=sk-你的key

    如果你用 OpenAI：
        LLM_BACKEND=openai
        OPENAI_API_KEY=sk-你的key

    如果你用 Claude：
        LLM_BACKEND=claude
        CLAUDE_API_KEY=sk-ant-你的key

填完保存，关闭。


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
第5步：启动！
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

在终端里确保当前在"创作系统"文件夹，然后：

    python main.py

看到彩色界面就成功了！输入 1 开始创作。


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
常见问题
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Q: pip install 报错"不是内部或外部命令"
A: Python 没装好。回到第2步，重装时记得勾"Add Python to PATH"

Q: 启动后报"API密钥未配置"
A: 回到第4步，确认 .env 文件已创建且 API Key 填写正确

Q: 启动后报"找不到 xxx 模块"
A: 回到第3步重跑 pip install

Q: 运行到一半报"TypeError: chat() takes X positional arguments"
A: 你可能用了旧版 Python（3.9以下），升级到 3.12

文件结构说明
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

创作系统/
├── main.py              ← 入口文件，双击或用 python main.py 启动
├── requirements.txt     ← 依赖清单（pip install 用）
├── .env                 ← 你的 API 密钥（需要自己创建）
├── .env.example         ← 密钥模板，参考它来创建 .env
├── workflow.yaml        ← 工作流配置（想改流程可以编辑它）
├── agents/              ← 各个 Agent 的代码（不用动）
├── core/                ← 框架核心代码（不用动）
├── llm/                 ← AI 模型调用（不用动）
├── prompts/             ← 提示词模板（可以按需修改）
├── tools/               ← 辅助工具（不用动）
├── projects/            ← 你的项目文件（重要！备份这个）
│   └── 悬疑电影/        ← 示例项目
├── 搬运指南.txt          ← 就是这个文件
└── README.md            ← 项目说明

提示：projects/ 文件夹是你的创作数据，建议单独备份。
